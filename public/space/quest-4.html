export function validateAdventureChoice(input: string): string {
  if (!input?.trim()) {
    throw new Error('Choice is required');
  }
  
  if (input.length > 200) {
    throw new Error('Choice too long (max 200 characters)');
  }
  
  return input.trim();
}

This ensures user input is properly sanitized, much like preventing noisy interference in data streams.

export function sanitizeForDisplay(input: string, maxLength = 1000): string {
  if (!input) return '';
  
  return input
    .replace(/[<>{}"`]/g, '')
    .replace(/\s+/g, ' ')
    .trim()
    .slice(0, maxLength);
}

Here, the function safeguards data for display, acting like a filtration system for visual panels on the Code Explorer.
