export function parseTheme(input: string): AdventureTheme | null {
  if (!input) return null;
  
  const normalized = input.trim().toLowerCase();
  
  const exactMatch = getThemeByKey(normalized);
  if (exactMatch) return exactMatch.key as AdventureTheme;
  
  const numericId = parseInt(normalized, 10);
  if (!isNaN(numericId)) {
    const byId = getThemeById(numericId);
    if (byId) return byId.key as AdventureTheme;
  }
  
  for (const theme of THEMES_ARRAY) {
    if (theme.keywords.some(keyword => normalized.includes(keyword.toLowerCase()))) {
      return theme.key as AdventureTheme;
    }
  }
  
  return null;
}

This illustrates how string normalization and matching patterns are used to parse user input into recognizable theme keys.

export function isValidTheme(theme: string): theme is AdventureTheme {
  return THEMES_ARRAY.some(t => t.key === theme);
}

A critical safeguard that ensures only defined themes are accepted during validation.

export function validateAdventureChoice(input: string): string {
  if (!input?.trim()) {
    throw new Error('Choice is required');
  }
  
  if (input.length > 200) {
    throw new Error('Choice too long (max 200 characters)');
  }
  
  return input.trim();
}

This enforces player input length and content rules to maintain mission-guided command limits.

export function validateTheme(input: string): string {
  if (!input?.trim()) {
    throw new Error('Theme is required');
  }
  
  const normalized = input.toLowerCase().trim();
  const parsedTheme = parseTheme(normalized);
  
  if (!parsedTheme || !isValidTheme(parsedTheme)) {
    const validThemes = getAllThemes().map(t => t.key).join(', ');
    throw new Error(`Invalid theme. Valid themes: ${validThemes}`);
  }
  
  return parsedTheme;
}

By leveraging theme parsing and validation functions, this ensures seamless integration with pre-configured themes.

export function sanitizeForDisplay(input: string, maxLength = 1000): string {
  if (!input) return '';
  
  return input
    .replace(/[<>{}"`]/g, '')
    .replace(/\s+/g, ' ')
    .trim()
    .slice(0, maxLength);
}

This strips potentially unsafe characters while preserving user-friendly output.
