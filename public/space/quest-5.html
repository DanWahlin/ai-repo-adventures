export function parseAdventureConfig(projectPath: string): unknown | null {
  const raw = loadAdventureConfig(projectPath);
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch {
    return null;
  }
}
  • This function serves as the central gateway for converting the raw configuration file (adventure.config.json) into a usable JavaScript object.
  • Error-handling mechanisms ensure that JSON syntax errors or missing files do not crash the system.
  • Separates file I/O operations (loadAdventureConfig) from parsing logic for cleaner responsibilities.
  • Enables downstream functions, like extractUniqueFilePaths, to work with structured configuration data.
export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];

  const unique = new Set<string>();
  const stack: any[] = [parsed];

  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;

    if (typeof node.path === 'string' && node.path.trim()) {
      const rel = node.path.trim();
      if (fs.existsSync(path.resolve(projectPath, rel))) {
        unique.add(rel);
      }
    }

    const children = Array.isArray(node) ? node : Object.values(node);
    children.forEach(child => {
      if (child && typeof child === 'object') stack.push(child);
    });
  }

  return Array.from(unique);
}
  • Implements a traversal mechanism to dynamically process nested configuration objects, ensuring scalability for large configurations.
  • Validates file paths with fs.existsSync to include only those present in the filesystem.
  • The use of Set prevents duplicate entries, ensuring uniqueness of extracted paths.
  • Demonstrates an iterative traversal pattern, which avoids recursion depth limits, a practical consideration for deeply nested JSON structures.
export function formatAdventureConfigForPrompt(projectPath: string): string {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') {
    return '';
  }

  const adventure = (parsed as any).adventure;
  if (!adventure || !Array.isArray(adventure.quests)) {
    return '';
  }

  let formatted = `## Quest Structure\n`;

  for (const quest of adventure.quests) {
    if (!quest.title || !Array.isArray(quest.files)) continue;

    formatted += `### ${quest.title}\n`;
    
    const filePaths = quest.files.map((f: any) => f.path).filter(Boolean);
    formatted += `Files: ${filePaths.join(', ')}\n`;
    
    const functions = quest.files
      .flatMap((f: any) => f.highlights || [])
      .map((h: any) => h.name)
      .filter(Boolean);
    formatted += `Functions: ${functions.join(', ')}\n\n`;
  }

  return formatted;
}
  • Reduces configuration size by stripping non-essential details while retaining crucial quest structure.
  • Converts nested quests and their files into a concise, human-readable format designed for LLM consumption.
  • Focuses on file paths and function highlights to ensure the output format supports purpose-driven, large-scale parsing workflows.
