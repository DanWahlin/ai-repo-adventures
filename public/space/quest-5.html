packages/core/src/analyzer/repo-analyzer.ts

export function generateRepomixContext(projectPath: string): string | null {
  const valid = validateProjectPath(projectPath);
  if (!valid) {
    throw new Error('Invalid project path');
  }  
  const command = `repomix analyze --path=${projectPath} --output=json`;  
  return captureRepomixStdout(command);  
}

This function is like a cosmic scanner, ensuring the safe mapping of territories before using a broader lens to capture their expansive details. 

export function captureRepomixStdout(command: string): string | null {
  try {
    // Execute CLI and capture structured terminus output
    const output = execSync(command, { encoding: 'utf-8' });  
    return output;
  } catch (err) {
    console.error(`Failed to capture Repomix output: ${err.message}`);
    return null;
  }
}

Imagine this as a cosmic transmission device that translates raw starship signals from the Nexus into decipherable star charts.
