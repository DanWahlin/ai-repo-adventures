async generateAllThemes(args: Map<string, string>): Promise<void> {

  const themePromises = themes.map(async (theme) => {

    themeGenerator['selectedTheme'] = theme;

  });

packages/generator/src/cli/template-engine.ts

renderPage(contentTemplate: string, variables: TemplateVariables): string {
  const baseTemplate = this.loadTemplate('base-template.html');
  const contentHtml = this.replacePlaceholders(this.loadTemplate(contentTemplate), variables);
  return this.replacePlaceholders(baseTemplate, { ...variables, CONTENT: contentHtml });
}
  • Combines modular templates (base-template, contentTemplate) for consistent design.
  • Injects dynamic content seamlessly while preserving separation of concerns.
  • Supports flexible variable insertion, accommodating various configurations.
private replacePlaceholders(template: string, variables: TemplateVariables): string {
  let result = template;
  for (const [key, value] of Object.entries(variables)) {
    const placeholder = `{{${key}}}`;
    result = result.replace(new RegExp(placeholder, 'g'), value || '');
  }
  return result;
}
  • Acts as a core utility for replacing placeholders with dynamic data.
  • Ensures proper matching and replacement through regex, supporting scalability.
  • Maintains cache integrity for repeated rendering tasks.

packages/generator/src/cli/asset-manager.ts

copyGlobalAssets(outputDir: string): void {
  const globalImages = ['global-icon.svg', 'shared-style.css'];
  globalImages.forEach(file => {
    const sourcePath = path.join(this.sourceDir, file);
    const targetPath = path.join(outputDir, file);
    if (fs.existsSync(sourcePath)) fs.copyFileSync(sourcePath, targetPath);
  });
}
  • Simplifies copying shared assets across themes, ensuring consistency.
  • Validates file paths to prevent errors during copying operations.
  • Introduces robust asset management for multi-theme compatibility.
copyQuestNavigator(outputDir: string): void {
  const templatesDir = path.join(this.sourceDir, 'templates');
  const assetsDir = path.join(outputDir, 'assets');
  const navigatorFiles = ['quest-navigator.css', 'quest-navigator.js'];
  navigatorFiles.forEach(file => {
    const sourcePath = path.join(templatesDir, file);
    fs.copyFileSync(sourcePath, path.join(assetsDir, file));
  });
}
  • Centralized handling of navigation-related assets for quests.
  • Guarantees compatibility for single-theme configurations.
  • Ensures linking files are correctly placed, improving UX and functionality.
