packages/mcp/src/server.ts

private setupHandlers() {
  this.server.setRequestHandler(ListToolsRequestSchema, async () => {
    const toolList = Object.entries(tools).map(([name, tool]) => ({
      name,
      description: tool.description,
      inputSchema: zodToJsonSchema(tool.schema, { 
        target: 'jsonSchema7',
        $refStrategy: 'none'
      })
    }));
    return { tools: toolList };
  });

  this.server.setRequestHandler(CallToolRequestSchema, async (request) => {
    try {
      const { name, arguments: args } = request.params;
      if (!(name in tools)) {
        throw new McpError(ErrorCode.MethodNotFound, `Unknown tool: ${name}`);
      }
      const tool = tools[name as keyof typeof tools];
      const validationResult = tool.schema.safeParse(args);
      if (!validationResult.success) {
        const errorMessages = validationResult.error.issues.map((err) => 
          `${err.path.join('.')}: ${err.message}`
        ).join(', ');
        throw new McpError(ErrorCode.InvalidParams, `Invalid parameters: ${errorMessages}`);
      }
      return await tool.handler(validationResult.data as any);
    } catch (error) {
      if (error instanceof McpError) {
        throw error;
      }
      throw new McpError(
        ErrorCode.InternalError,
        `Tool execution failed: ${error instanceof Error ? error.message : String(error)}`
      );
    }
  });
}
  • Dynamically maps tool handlers to schemas (ListToolsRequestSchema, CallToolRequestSchema) for structured functionality.
  • Uses zodToJsonSchema to convert schemas into JSON-friendly formats for consumption.
  • Implements robust validation using Zod, returning actionable error messages when input deviations occur.
  • Handles execution flow errors through custom McpError, distinguishing user and server-level issues.
async run() {
  const transport = new StdioServerTransport();
  await this.server.connect(transport);
  console.error('Repo Adventure MCP server running on stdio');
  
  const projectPath = process.cwd();
  console.error(`Pre-generating repomix content for project at ${projectPath}...`);
  repoAnalyzer.preGenerate(projectPath);
}
  • Establishes server operations over StdioServerTransport, ensuring compatibility with standard streams.
  • Logs server activation messages, enhancing situational awareness for system operators.
  • Prepares repository analysis with repoAnalyzer.preGenerate, warming cache for optimized performance during runtime.
async function main() {
  try {
    const server = new RepoAdventureServer();
    ['SIGINT', 'SIGTERM'].forEach(sig => 
      process.on(sig as NodeJS.Signals, gracefulShutdown)
    );
    process.on('unhandledRejection', (reason) => {
      console.error('Unhandled promise rejection:', reason);
      console.error('MCP server continuing to run. Please report this error.');
    });
    await server.run();
  } catch (error) {
    console.error('Fatal error starting MCP server:', error);
    process.exit(1);
  }
}

function gracefulShutdown() {
  console.error('\nShutting down MCP server...');
  try {
    repoAnalyzer.cleanup();
  } catch (e) {
    console.error('Cleanup error:', e);
  }
  process.exit(0);
}
  • Ensures graceful system exit with signal (SIGINT, SIGTERM) handlers invoking gracefulShutdown.
  • Captures asynchronous errors through the unhandledRejection listener, maintaining operational continuity.
  • Implements cleanup logic with repoAnalyzer.cleanup, providing a robust finalization process to release resources.
