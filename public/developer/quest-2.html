async initializeAdventure(
  projectInfo: ProjectInfo, 
  theme: AdventureTheme, 
  projectPath?: string,
  customThemeData?: CustomThemeData
): Promise<string> {
  // Reset state for new adventure
  this.state.reset();
  this.state.projectInfo = projectInfo;
  this.state.currentTheme = theme;
  this.state.projectPath = projectPath || process.cwd();
  
  // Generate the overall story and quests using LLM
  const storyResponse = await this.storyGenerator.generateStoryAndQuests(projectInfo, theme, this.state.projectPath);
  
  this.state.title = storyResponse.title;
  this.state.story = storyResponse.story;
  
  this.state.quests = this.enforceConfigQuestCount(storyResponse.quests, this.state.projectPath);
  
  // Return the story with available quests
  return this.formatStoryWithQuests({
    ...storyResponse,
    quests: this.state.quests
  });
}
  • Initializes a new adventure by resetting state, setting the theme, and generating the story/quests.
  • Invokes the StoryGenerator to produce narrative content based on the project's context and theme.
  • Enforces quest configurations by referencing adventure.config.json, guaranteeing alignment with predefined limits.
  • Formats the result, combining the story and the list of quests into a structured output.
async exploreQuest(choice: string): Promise<AdventureResult> {
  const sanitizedChoice = this.validateAndSanitizeChoice(choice);
  
  const quest = this.findQuest(sanitizedChoice);
  if (!quest) {
    return this.createNotFoundResult();
  }

  return await this.executeQuest(quest);
}
  • Interprets user input for quest selection, sanitizing and mapping it to a specific quest object.
  • Fallbacks are applied if the desired quest is not found to guide users back to valid choices.
  • Calls executeQuest to load or generate quest content, enabling efficient exploration.
get progressPercentage(): number {
  return this.quests.length > 0 
    ? Math.round((this.completedQuests.size / this.quests.length) * 100)

}
