async initializeAdventure(
  projectInfo: ProjectInfo, 
  theme: AdventureTheme, 
  projectPath?: string,
  customThemeData?: CustomThemeData
): Promise<string> {
  this.state.reset();
  this.state.projectInfo = projectInfo;
  this.state.currentTheme = theme;
  this.state.projectPath = projectPath || process.cwd();

  if (theme === 'custom' && customThemeData) {
    this.storyGenerator.setCustomTheme(customThemeData);
  }

  const storyResponse = await this.storyGenerator.generateStoryAndQuests(projectInfo, theme, this.state.projectPath);
  this.state.title = storyResponse.title;
  this.state.story = storyResponse.story;
  this.state.quests = this.enforceConfigQuestCount(
    this.mergeQuestFilesFromConfig(storyResponse.quests, this.state.projectPath), 
    this.state.projectPath
  );

  return this.formatStoryWithQuests({ ...storyResponse, quests: this.state.quests });
}
  • Initializes quests by resetting the state, applying themes, and validating input.
  • Combines project info with story and quest content generated by StoryGenerator.
  • Enforces configuration rules to filter and merge quests dynamically.

async generateStoryAndQuests(
  projectInfo: ProjectInfo, 
  theme: AdventureTheme, 
  projectPath?: string
): Promise<StoryResponse> {
  const validatedTheme = this.validateTheme(theme);
  const response = await this.withTimeout(
    this.llmClient.generateResponse(
      loadStoryGenerationPrompt({
        theme: validatedTheme, 
        ...(formatAdventureConfigForPrompt(projectPath) && { adventureGuidance: formatAdventureConfigForPrompt(projectPath) }),
        customInstructions: extractCustomInstructions(projectPath) || '',
        repomixContent: projectInfo.repomixContent || 'No repo content available'
      }), 
      { maxTokens: LLM_MAX_TOKENS_STORY }
    )
  );

  return parseMarkdownToStoryResponse(response.content.trim());
}
  • Fuses LLM generation with custom configuration inputs for contextual storytelling.
  • Leverages Markdown parsing to convert raw LLM outputs into structured readable data.
async generateQuestContent(config: QuestGenerationConfig): Promise<QuestContent> {
  const response = await this.withTimeout(
    this.llmClient.generateResponse(
      loadQuestContentPrompt({ 
        theme: config.theme,
        adventureTitle: config.quest.title,
        codeContent: config.codeContent,
        storyContent: this.currentStoryContent || 'No story available.',
        ...(config.questPosition && { questPosition: config.questPosition }),
        totalQuests: config.totalQuests || 0
      }),
      { maxTokens: LLM_MAX_TOKENS_QUEST }
    )
  );

  return { adventure: response.content.trim(), fileExploration: '', codeSnippets: [], hints: [] };
}
  • Synthesizes quest-specific content with a focus on code exploration and theme adherence.
  • Parses responses to maintain compatibility between adventure generation and story prompts.
function parseMarkdownToStoryResponse(markdownContent: string): StoryResponse {
  const tokens = marked.lexer(markdownContent);
  const title = extractTitle(tokens);
  let story = '';
  const quests: Quest[] = [];
  let currentQuest: Partial<Quest> = {};

  for (const token of tokens) {
    if (token.type === 'heading' && token.depth === 1) {
      currentQuest.title = token.text;
    } else if (token.type === 'paragraph') {
      story += token.text;
    } else if (token.type === 'list') {
      currentQuest.codeFiles = currentQuest.codeFiles || [];
      token.items.forEach((item: any) => currentQuest.codeFiles?.push(item.text));
    }
  }
  
  return { title, story, quests };
}
  • Extracts structured adventure information (title, story, quests) from markdown tokens.
  • Distills textual data for further processing within the quest pipeline.

export function parseAdventureConfig(projectPath: string): unknown | null {
  const raw = loadAdventureConfig(projectPath);
  if (!raw) return null;
  
  try {
    return JSON.parse(raw);
  } catch {
    return null;
  }
}
  • Provides a single point of parsing for adventure.config.json files, ensuring error tolerance.
  • Protects against invalid inputs but still leverages configuration versatility.
export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  const unique = new Set<string>();

  const stack: any[] = [parsed];
  while (stack.length) {
    const node = stack.pop();
    if (node?.path) {
      const fullPath = path.resolve(projectPath, node.path);
      if (fs.existsSync(fullPath)) {
        unique.add(node.path);
      }
    }

    if (Array.isArray(node) || typeof node === 'object') {
      stack.push(...Object.values(node));
    }
  }
  return Array.from(unique);
}
  • Gathers all referenced paths from configuration for validation, ensuring repo-wide consistency.
  • Prevents unneeded file analysis by eliminating duplicates upfront.
export function formatAdventureConfigForPrompt(projectPath: string): string {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return '';

  const adventure = parsed.adventure?.quests || [];
  return adventure.map((q: any) => `### ${q.title}\nFiles: ${q.files.map((f: any) => f.path).join(', ')}\n`).join('\n');
}
  • Prepares compact configuration for LLM prompt environments, enhancing efficiency.
  • Focuses solely on relevant, actionable quest content.
