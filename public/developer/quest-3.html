packages/core/src/adventure/adventure-manager.ts: Quest and State Management

This file contains the AdventureManager class that initializes adventures, handles quest execution, and manages the state of ongoing adventures. It connects directly with other subsystems like StoryGenerator and adventure-config.

Highlights

  • initializeAdventure: Clears the current adventure state, sets a theme, and generates stories and quests via the StoryGenerator.
  • exploreQuest: Executes selected quests, validates user input, and retrieves quest content from either cache or a new generation process.
  • generateAdventureContent: Prepares code content (via file paths or full repository data) for LLM-based quest generation.
  • progressPercentage: Calculates overall quest progress as a percentage of completion.

async initializeAdventure(
  projectInfo: ProjectInfo, 
  theme: AdventureTheme, 
  projectPath?: string,
  customThemeData?: CustomThemeData
): Promise<string> {
  // Reset state for new adventure
  this.state.reset();
  this.state.projectInfo = projectInfo;
  this.state.currentTheme = theme;
  this.state.projectPath = projectPath || process.cwd();
  
  // Set custom theme data if provided
  if (theme === 'custom' && customThemeData) {
    this.storyGenerator.setCustomTheme(customThemeData);
  }

  // Generate the overall story and quests using LLM
  const storyResponse = await this.storyGenerator.generateStoryAndQuests(
    projectInfo, theme, this.state.projectPath
  );
  this.state.title = storyResponse.title;
  this.state.story = storyResponse.story;

  return this.formatStoryWithQuests({
    ...storyResponse,
    quests: this.state.quests
  });
}
  • Initializes a clean slate for a new adventure, resetting all state variables.
  • Integrates project context and user preferences to personalize the adventure.
  • Bridges StoryGenerator to fetch adventure narratives and quests.
  • Returns a formatted journey overview constructed from generated content.
async exploreQuest(choice: string): Promise<AdventureResult> {
  const sanitizedChoice = this.validateAndSanitizeChoice(choice);
  const quest = this.findQuest(sanitizedChoice);
  if (!quest) {
    return this.createNotFoundResult();
  }
  
  return await this.executeQuest(quest);
}
  • Validates user choices and maps them to a specific quest.
  • Leverages caching for completed quests, reducing repetitive calls to StoryGenerator.
  • Ensures proper execution or assists users if their choice is invalid.
get progressPercentage(): number {
  return this.quests.length > 0

async generateQuestContent(config: QuestGenerationConfig): Promise<QuestContent> {
  const { quest, theme, codeContent, questPosition, totalQuests } = config;

  const prompt = loadQuestContentPrompt({
    theme,
    adventureTitle: quest.title,
    codeContent,
    storyContent: this.currentStoryContent || 'No story context available.',
    questPosition,
    totalQuests
  });

  const response = await this.withTimeout(
    this.llmClient.generateResponse(prompt, { maxTokens: LLM_MAX_TOKENS_QUEST })
  );

  return {
    adventure: response.content.trim(),
    fileExploration: '',
    codeSnippets: [],
    hints: []
  };
}
  • Maps configuration inputs into quest-specific markdown outputs through LLMs.
  • Combines adventure guidance and project metadata to enrich the content generation process.
  • Provides insights based on file/user requirements beyond generic templates.
private validateTheme(theme: AdventureTheme): AdventureTheme {
  if (!isValidTheme(theme)) {
    console.warn(`Invalid theme '${theme}', defaulting to ${DEFAULT_THEME}`);
    return DEFAULT_THEME;
  }
  return theme;
}
  • Ensures themes are valid and safe for use during the generation pipeline.
  • Protects system stability by falling back to default themes when necessary.

export function loadAdventureConfig(projectPath: string): string | null {
  const configPath = path.join(projectPath, ADVENTURE_CONFIG_FILE);
  return readFileIfExists(configPath);
}
  • Executes a safe read operation to load raw configuration data.
  • Prevents blocking or crash issues in the absence of a config file.
export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];

  const unique = new Set<string>();
  const stack: any[] = [parsed];

  while (stack.length) {
    const node = stack.pop();
    if (typeof node.path === 'string') {
      const relPath = node.path.trim();
      if (fs.existsSync(path.resolve(projectPath, relPath))) {
        unique.add(relPath);
      }
    }
    const children = Array.isArray(node) ? node : Object.values(node);
    stack.push(...children.filter(child => typeof child === 'object'));
  }

  return Array.from(unique);
}
  • Recursively parses configuration files to identify and validate project paths.
  • Facilitates LLM-targeted content generation by isolating relevant files.
export function formatAdventureConfigForPrompt(projectPath: string): string {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed) return '';

  return `## Quest Structure\nQuests:\n${parsed.adventure?.quests.map((q: any) => q.title).join(', ')}`;
}
  • Converts verbose configuration data into a compact, LLM-appropriate format.
  • Enhances content generation efficiency by removing unnecessary complexity.
