packages/generator/src/cli/theme-manager.ts

generateThemeCSS(theme: AdventureTheme, outputDir: string): void {
  const cssFiles = [
    path.join(this.themesDir, 'base.css'),
    path.join(this.themesDir, `${theme}.css`)
  ];

  let combinedCSS = '';
  cssFiles.forEach(cssFile => {
    if (fs.existsSync(cssFile)) {
      combinedCSS += fs.readFileSync(cssFile, 'utf8') + '\n';
    }
  });
  
  const cssPath = path.join(outputDir, 'assets', 'theme.css');
  fs.writeFileSync(cssPath, combinedCSS);
}
  • What this code does: Combines theme-specific CSS files into a single stylesheet for each adventure.
  • Patterns or techniques: Uses file system utilities (fs.existsSync, fs.readFileSync) for dynamic asset management.
  • Why this approach was chosen: Ensures consistency and minimizes duplication in CSS generation.
  • Connection to other components: Generated CSS is used during quest presentation and navigation.
  • Details to notice: CSS files include both base styles and specific theme styles for versatility.
