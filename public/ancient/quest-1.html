/**
 * Load a template file with caching
 */
private loadTemplate(templateName: string): string {
  if (this.templateCache.has(templateName)) {
    return this.templateCache.get(templateName)!;
  }

  const templatePath = path.join(this.templatesDir, templateName);
  
  try {
    const content = fs.readFileSync(templatePath, 'utf-8');
    this.templateCache.set(templateName, content);
    return content;
  } catch (error) {
    throw new Error(`Failed to load template ${templateName}: ${error}`);
  }
}
  • Manages loading HTML templates into memory for use.
  • Implements caching to reduce redundant reads from the file system.
  • Caches templates in the templateCache map to ensure efficiency.
/**
 * Replace placeholders in the template with variables
 */
private replacePlaceholders(template: string, variables: TemplateVariables): string {
  let result = template;

  for (const [key, value] of Object.entries(variables)) {
    const placeholder = `{{${key}}}`;
    result = result.replace(new RegExp(placeholder, 'g'), value || '');
  }

  return result;
}
  • Replaces {{placeholder}} tokens in templates dynamically at runtime.
  • Ensures each placeholder is substituted with corresponding data from the provided variables.
  • Demonstrates the use of regex to enforce flexibility with multiple replacements.
/**
 * Render a complete page using base template + content template
 */
renderPage(contentTemplate: string, variables: TemplateVariables): string {
  const baseTemplate = this.loadTemplate('base-template.html');
  const contentHtml = this.loadTemplate(contentTemplate);
  
  const renderedContent = this.replacePlaceholders(contentHtml, variables);
  
  const pageVariables = {
    ...variables,
    CONTENT: renderedContent
  };

  return this.replacePlaceholders(baseTemplate, pageVariables);
}
  • Combines the main structure of the page (base template) with specific content (content template).
  • Utilizes replacePlaceholders to ensure templating remains dynamic and reusable.
  • Allows for the layering of templates, reflecting the ancient rituals of structured storytelling.

private buildQuestHTML(quest: QuestInfo, content: string, questIndex: number): string {
  const prevQuest = questIndex > 0 ? this.quests[questIndex - 1] : null;
  const nextQuest = questIndex < this.quests.length - 1 ? this.quests[questIndex + 1] : null;

  let bottomNavigation = '';
  if (prevQuest || nextQuest) {
    bottomNavigation = `
    <div class="quest-navigation">
      ${prevQuest ? `<a href="${prevQuest.filename}">‚Üê Previous</a>` : ''}
      ${nextQuest ? `<a href="${nextQuest.filename}">Next ‚Üí</a>` : ''}
    </div>`;
  }

  const variables = {
    PAGE_TITLE: this.stripHTML(quest.title),
    QUEST_CONTENT: this.formatMarkdown(content),
    BOTTOM_NAVIGATION: bottomNavigation
  };

  return this.templateEngine.renderPage('quest-template.html', variables);
}
  • Creates the HTML structure for a single quest page with navigation to prior and next quests.
  • Integrates dynamic placeholders for title and content using TemplateEngine.renderPage.
  • Includes guards to gracefully skip non-existent navigation.
async startHttpServer(port: number = 8080): Promise<void> {
  const server = http.createServer((req, res) => {
    let filePath = path.join(this.outputDir, req.url === '/' ? 'index.html' : req.url || '');
    fs.readFile(filePath, (err, content) => {
      if (err) {
        res.writeHead(err.code === 'ENOENT' ? 404 : 500);
        res.end('Server error');
      } else {
        res.writeHead(200, { 'Content-Type': 'text/html' });
        res.end(content, 'utf-8');
      }
    });
  });

  server.listen(port, () => {
    console.log(`Server running on http://localhost:${port}`);
  });
}
  • Launches a local HTTP server to serve the static quest site content.
  • Provides developers an accessible interface to view templates as crafted chronicles.

copyGlobalAssets(outputDir: string): void {
  try {
    const globalAssetsDir = path.join(outputDir, 'assets');
    fs.mkdirSync(globalAssetsDir, { recursive: true });

    // Copy shared images
    const globalImages = ['github-mark.svg', 'github-mark-white.svg'];
    globalImages.forEach(file => {
      const sourcePath = path.join(this.sourceDir, 'assets', 'shared', file);
      if (fs.existsSync(sourcePath)) {
        const targetPath = path.join(globalAssetsDir, file);
        fs.copyFileSync(sourcePath, targetPath);
      }
    });

  } catch (error) {
    console.log('‚ö†Ô∏è Warning: Could not copy global assets');
  }
}
