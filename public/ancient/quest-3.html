export function parseAdventureConfig(projectPath: string): unknown | null {
  const raw = loadAdventureConfig(projectPath);
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch {
    return null;
  }
}
  • Converts the raw knowledge text (adventure.config.json) into a usable data structure.
  • Leverages try-catch blocks to handle errors from malformed or missing files, ensuring no catastrophic interruption of quest generation.
  • Simplifies data access for other storytelling mechanisms by validating the file format.
export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];

  const unique = new Set<string>();
  const stack: any[] = [parsed];

  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;

    if (typeof node.path === 'string' && node.path.trim()) {
      const rel = node.path.trim();
      if (fs.existsSync(path.resolve(projectPath, rel))) {
        unique.add(rel);
      }
    }

    const children = Array.isArray(node) ? node : Object.values(node);
    children.forEach(child => {
      if (child && typeof child === 'object') stack.push(child);
    });
  }

  return Array.from(unique);
}
  • Traverses the parsed configuration to unearth unique file paths critical for storytelling.
  • Avoids redundancy by using a Set, ensuring each file is added only once.
  • Validates file existence, highlighting a focus on practicality and precision during ancient knowledge integration.

async generateQuestContent(config: QuestGenerationConfig): Promise<QuestContent> {
  const { quest, theme, codeContent, questPosition, totalQuests } = config;
  let adventureGuidance = '';
  let customInstructions = '';
  
  if (this.projectPath) {
    const formattedConfig = formatAdventureConfigForPrompt(this.projectPath);
    if (formattedConfig) {
      adventureGuidance = formattedConfig;
    }

    const customInstructionsFromConfig = extractCustomInstructions(this.projectPath);
    if (customInstructionsFromConfig) {
      customInstructions = customInstructionsFromConfig;
    }
  }

  const prompt = loadQuestContentPrompt({
    theme,
    adventureTitle: quest.title,
    codeContent,
    adventureGuidance: adventureGuidance || '',
    ...(customInstructions && { customInstructions })
  });

  const response = await this.withTimeout(
    this.llmClient.generateResponse(prompt, { maxTokens: LLM_MAX_TOKENS_QUEST })
  );

  if (!response.content || response.content.trim() === '') {
    throw new Error('LLM returned empty response for quest content');
  }

  return { adventure: response.content.trim(), codeSnippets: [], hints: [] };
}

function parseMarkdownToStoryResponse(markdownContent: string): StoryResponse {
  const tokens = marked.lexer(markdownContent);
  const title = extractTitle(tokens);
  let story = '';
  const quests: Quest[] = [];
  
  let currentSection = '';
  let currentQuest: Partial<Quest> = {};

  for (const token of tokens) {
    if (token.type === 'heading') {
      if (token.depth === 1) {
        currentSection = 'story';
      } else if (token.depth === 2) {
        currentSection = token.text.toLowerCase();
      }
    } else if (token.type === 'paragraph') {
      if (currentSection === 'story') {
        story += token.text + '\n\n';
      }
    } else if (token.type === 'list' && currentQuest.title) {
      extractCodeFilesFromList(token, currentQuest);
    }
  }

  return { title, story: story.trim(), quests };
}
  • Parses the markdown input into readable structures like title, story, and quests.
  • Divides content into sections for clarity, ensuring each quest is isolated and formatted correctly.
  • Establishes connections between markdown-based storytelling and structured code-driven narratives.
