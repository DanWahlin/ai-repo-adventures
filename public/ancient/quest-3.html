packages/core/src/shared/theme.ts

/**
 * Adventure Themes System: Defines accepted themes for generated adventures
 */

export const THEMES = {
  ancient: "Ancient ruins, forgotten wisdom, pyramids",
  space: "Cosmic exploration, distant galaxies, alien worlds",
  jungle: "Treacherous forests, hidden civilizations, survival",
};

export function validateThemeInput(input: string): string {
  const normalizedInput = input.toLowerCase().trim();
  if (!THEMES[normalizedInput]) {
    throw new Error(`Invalid theme specified: ${input}. Choose from ${Object.keys(THEMES).join(', ')}`);
  }
  return normalizedInput;
}

export function getAllThemes(): string[] {
  return Object.keys(THEMES).map(themeKey => THEMES[themeKey]);
}

The THEMES object acts as a sacred lexicon of possible landscapes for adventurers. validateThemeInput is akin to a gatekeeper, ensuring no unworthy or unrecognized themes destabilize adventures.
