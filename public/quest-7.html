export function validateAdventureChoice(input: string): string {
  if (!input?.trim()) {
    throw new Error('Choice is required');
  }

  if (input.length > 200) {
    throw new Error('Choice too long (max 200 characters)');
  }

  return input.trim();
}

🔍 Real-World Analogy: This function acts like the entry scanner for a starship docking bay—only admitting ships that meet certain criteria. Just as a scanner checks for proper credentials and scans dimensions to ensure safety, validateAdventureChoice ensures user inputs are non-empty, concise, and properly formatted. This reduces the risk of errors disrupting critical operational systems.

export function validateTheme(input: string): string {
  if (!input?.trim()) {
    throw new Error('Theme is required');
  }

  const normalized = input.toLowerCase().trim();
  const parsedTheme = parseTheme(normalized);

  if (!parsedTheme || !isValidTheme(parsedTheme)) {
    const validThemes = getAllThemes().map(t => t.key).join(', ');
    throw new Error(`Invalid theme. Valid themes: ${validThemes}`);
  }

  return parsedTheme;
}

🔍 Real-World Analogy: This function is akin to the mission control desk verifying the legitimacy of a protocol before initiating a launch. By trimming and normalizing the theme name, and cross-referencing it with a list of valid themes, the function prevents unauthorized commands (like invalid themes) from being used, safeguarding mission consistency and galactic security.
