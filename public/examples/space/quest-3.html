export function loadAdventureConfig(projectPath: string): string | null {
  const configPath = path.join(projectPath, ADVENTURE_CONFIG_FILE);
  return readFileIfExists(configPath);
}
  • Reads the adventure.config.json file from the specified project path.
  • Provides a pure file read operation, returning raw text or null if the file is missing.
  • Ensures the system gracefully handles missing configurations without failure.
  • Serves as the foundation for subsequent parsing and validation processes.
export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];

  const unique = new Set<string>();
  const stack: any[] = [parsed];

  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;

    if (typeof node.path === 'string' && node.path.trim()) {
      const rel = node.path.trim();
      if (fs.existsSync(path.resolve(projectPath, rel))) {
        unique.add(rel);
      }
    }

    const children = Array.isArray(node) ? node : Object.values(node);
    children.forEach(child => {
      if (child && typeof child === 'object') stack.push(child);
    });
  }

  return Array.from(unique);
}
  • Extracts all unique file paths referenced by path fields in the configuration.
  • Validates the existence of each file to ensure references are actionable.
  • Uses a depth-first traversal approach to handle nested configuration structures.
  • Enhances quest generation by providing a reliable list of target files.
export function formatAdventureConfigForPrompt(projectPath: string): string {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') {
    return '';
  }

  const adventure = (parsed as any).adventure;
  if (!adventure || !Array.isArray(adventure.quests)) {
    return '';
  }

  let formatted = `## Quest Structure\n`;

  for (const quest of adventure.quests) {
    if (!quest.title || !Array.isArray(quest.files)) continue;

    formatted += `### ${quest.title}\n`;
    const filePaths = quest.files.map((f: any) => f.path).filter(Boolean);
    formatted += `Files: ${filePaths.join(', ')}\n`;
    const functions = quest.files
      .flatMap((f: any) => f.highlights || [])
      .map((h: any) => h.name)
      .filter(Boolean);
    formatted += `Functions: ${functions.join(', ')}\n\n`;
  }

  return formatted;
}
  • Formats adventure configuration into a compact structure suitable for LLM prompts.
  • Eliminates redundant descriptions to reduce token usage while retaining essential details.
  • Provides a clear mapping of quests to associated files and functions.
  • Enables efficient quest generation by streamlining LLM input.
export function extractMicrosoftClarityCode(projectPath: string): string | null {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') {
    return null;
  }

  const adventure = (parsed as any).adventure;
  if (!adventure || typeof adventure.microsoftClarityCode !== 'string') {
    return null;
  }

  return adventure.microsoftClarityCode.trim() || null;
}
  • Extracts Microsoft Clarity tracking codes from the configuration file.
  • Enables conditional injection of analytics scripts into generated HTML adventures.
  • Provides a streamlined method for integrating user behavior tracking.
