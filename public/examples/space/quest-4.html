async generateRepomixContext(projectPath: string, options: RepomixOptions = {}): Promise<string> {
  this.validateProjectPath(projectPath);
  
  const configuredFiles = extractUniqueFilePaths(projectPath);
  if (configuredFiles.length > 0) {
    try {
      return await this.generateOptimizedContent(projectPath, configuredFiles);
    } catch (error) {
      console.warn(`Failed to generate optimized content, falling back to targeted content: ${error.message}`);
    }
  }
  
  console.log('No adventure.config.json found: analyzing full codebase (compressed)');
  const cacheKey = `${path.resolve(projectPath)}:${JSON.stringify(options)}`;
  const cached = this.cache.get(cacheKey);
  if (cached && (Date.now() - cached.timestamp) < REPOMIX_CACHE_TTL) {
    return cached.content;
  }
  
  try {
    const cliOptions: CliOptions = {
      style: options.style || 'markdown',
      compress: options.compress !== false,
      ignore: ['node_modules', 'dist', 'build', '.git'],
      removeComments: true,
      removeEmptyLines: true,
      noDirectoryStructure: true
    };
    const context = await this.captureRepomixStdout(['.'], projectPath, cliOptions);
    this.cache.set(cacheKey, { content: context, timestamp: Date.now() });
    return context;
  } catch (error) {
    throw new Error(`Repomix execution failed: ${error.message}`);
  }
}
  • This code validates the project path and checks for a pre-configured list of files to analyze.
  • It uses caching to avoid redundant computations, improving performance.
  • The fallback mechanism ensures the system remains functional even if optimized content generation fails.
  • The CliOptions object demonstrates the use of configurable parameters for the Repomix tool.
  • The method enforces resource limits through the captureRepomixStdout function.
async generateTargetedContent(projectPath: string, targetFiles: string[], compress: boolean = true): Promise<string> {
  this.validateProjectPath(projectPath);
  
  const safeFiles = this.validateAndNormalizeTargetFiles(projectPath, targetFiles);
  if (safeFiles.length === 0) {
    throw new Error('No valid target files found after validation');
  }
  
  const cacheKey = `${path.resolve(projectPath)}:targeted:${safeFiles.join(',')}:compress=${compress}`;
  const cached = this.cache.get(cacheKey);
  if (cached && (Date.now() - cached.timestamp) < REPOMIX_CACHE_TTL) {
    return cached.content;
  }
  
  try {
    const cliOptions: CliOptions = {
      style: 'markdown',
      compress,
      include: safeFiles.join(','),
      removeComments: compress,
      removeEmptyLines: compress,
      noDirectoryStructure: true
    };
    const context = await this.captureRepomixStdout(['.'], projectPath, cliOptions);
    this.cache.set(cacheKey, { content: context, timestamp: Date.now() });
    return context;
  } catch (error) {
    throw new Error(`Targeted repomix execution failed: ${error.message}`);
  }
}
  • This method targets specific files for analysis, reducing unnecessary processing.
  • It validates and normalizes file paths to ensure security and reliability.
  • Caching is implemented to store results for repeated queries.
  • The CliOptions configuration demonstrates how to tailor Repomix's behavior for targeted analysis.
  • The approach balances precision and efficiency, focusing on relevant files.

async generateResponse(prompt: string, options?: LLMRequestOptions): Promise<LLMResponse> {
  await this.applyThrottling();

  try {
    const requestParams = this.buildRequestParams(prompt, options);
    const completion = await this.executeRequest(requestParams);
    let content = this.validateResponse(completion);

    if (options?.responseFormat === 'json_object') {
      content = this.cleanJsonResponse(content);
    }

    this.logTokenUsage(completion, options?.context);
    this.onSuccessfulRequest();
    return { content };
  } catch (error) {
    const rateLimitInfo = this.detectRateLimitType(error);

    if (rateLimitInfo.type !== RateLimitType.NONE) {
      if (rateLimitInfo.type === RateLimitType.REQUEST_RATE_LIMIT) {
        this.activateThrottling(error);
      }
      throw new RateLimitError(rateLimitInfo.type, rateLimitInfo.waitSeconds, rateLimitInfo.message, error);
    }

    this.logDetailedError(error, prompt);
    throw new Error(`LLM request failed: ${error.message}`);
  }
}
  • This function demonstrates the complete lifecycle of an AI request, including throttling, execution, and error handling.
  • It validates responses to ensure they meet the expected format and handles empty or truncated outputs gracefully.
  • The method implements adaptive throttling to manage rate limits dynamically.
  • The use of structured logging enhances observability and debugging.
  • The RateLimitError class encapsulates detailed information about rate limit issues.
constructor() {
  this.model = LLM_MODEL;
  
  const apiKey = this.getApiKey();
  if (!apiKey || !LLM_BASE_URL) {
    throw new Error('LLM configuration required. Please set LLM_BASE_URL and appropriate API key.');
  }

  if (this.isAzureOpenAI()) {
    const azureEndpoint = LLM_BASE_URL.split('/openai/deployments')[0];
    this.client = new AzureOpenAI({
      endpoint: azureEndpoint,
      apiKey,
      apiVersion: LLM_API_VERSION,
      deployment: this.model
    });
  } else {
    this.client = new OpenAI({
      apiKey,
      baseURL: LLM_BASE_URL
    });
  }
}

  • It validates the presence of required environment variables and throws errors for missing configurations.
  • The design supports multiple AI providers, showcasing flexibility and extensibility.
  • The use of environment variables enhances security by avoiding hardcoded credentials.
  • The isAzureOpenAI method demonstrates provider-specific logic for initialization.
