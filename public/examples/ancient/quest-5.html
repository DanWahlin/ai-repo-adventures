export function parseTheme(input: string): AdventureTheme | null {
  if (!input) return null;
  const normalized = input.trim().toLowerCase();
  const exactMatch = getThemeByKey(normalized);
  if (exactMatch) return exactMatch.key as AdventureTheme;
  const numericId = parseInt(normalized, 10);
  if (!isNaN(numericId)) {
    const byId = getThemeById(numericId);
    if (byId) return byId.key as AdventureTheme;
  }
  for (const theme of THEMES_ARRAY) {
    if (theme.keywords.some(keyword => normalized.includes(keyword.toLowerCase()))) {
      return theme.key as AdventureTheme;
    }
  }
  return null;
}
  • Implements flexible parsing logic, supporting theme selection by ID, key, or keywords.
  • Uses normalization to handle case-insensitive and whitespace-tolerant input.
  • Demonstrates a fallback mechanism to match user input against theme keywords.

generateThemeCSS(theme: AdventureTheme, outputDir: string): void {
  const cssFiles = [
    path.join(this.themesDir, 'base.css'),
    path.join(this.themesDir, 'homepage.css'), 
    path.join(this.themesDir, 'animations.css'),
    path.join(this.themesDir, `${theme}.css`)
  ];
  let combinedCSS = '';
  cssFiles.forEach(cssFile => {
    if (fs.existsSync(cssFile)) {
      combinedCSS += fs.readFileSync(cssFile, 'utf8') + '\\n\\n';
    }
  });
  const cssPath = path.join(outputDir, 'assets', 'theme.css');
  fs.writeFileSync(cssPath, combinedCSS);
}
  • Dynamically combines CSS files to generate a theme-specific stylesheet.
  • Ensures modularity by separating base styles, animations, and theme-specific styles.
  • Writes the final CSS file to the appropriate output directory for use in static sites.

export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];
  const unique = new Set<string>();
  const stack: any[] = [parsed];
  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;
    if (typeof node.path === 'string' && node.path.trim()) {
      const rel = node.path.trim();
      if (fs.existsSync(path.resolve(projectPath, rel))) {
        unique.add(rel);
      }
    }
    const children = Array.isArray(node) ? node : Object.values(node);
    children.forEach(child => {
      if (child && typeof child === 'object') stack.push(child);
    });
  }
  return Array.from(unique);
}
  • Extracts and validates file paths from the configuration, ensuring they exist in the project.
  • Uses a depth-first traversal to handle nested configuration structures.
  • Demonstrates the use of Set to ensure uniqueness of extracted file paths.
