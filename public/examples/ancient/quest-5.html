export function getThemeById(id: number): ThemeDefinition | null {
  return THEMES_ARRAY.find(theme => theme.id === id) || null;
}

export function isValidTheme(theme: string): theme is AdventureTheme {
  return THEMES_ARRAY.some(t => t.key === theme);
}

export function parseTheme(input: string): AdventureTheme | null {
  if (!input) return null;
  
  const normalized = input.trim().toLowerCase();
  const exactMatch = getThemeByKey(normalized);
  if (exactMatch) return exactMatch.key as AdventureTheme;
  
  const numericId = parseInt(normalized, 10);
  if (!isNaN(numericId)) {
    const byId = getThemeById(numericId);
    if (byId) return byId.key as AdventureTheme;
  }
  
  for (const theme of THEMES_ARRAY) {
    if (theme.keywords.some(keyword => normalized.includes(keyword.toLowerCase()))) {
      return theme.key as AdventureTheme;
    }
  }
  
  return null;
}
  • getThemeById ensures seamless integration of numeric IDs with the theme system.
  • isValidTheme uses a type guard to ensure type safety and prevent invalid theme usage.
  • parseTheme demonstrates robust input handling by supporting multiple matching strategies.
  • The inclusion of keyword matching enhances user experience by allowing flexible inputs.
  • These functions collectively ensure that the theme system is both user-friendly and secure.
export function getFormattedThemeOptions(): string {
  return getAllThemes()

parseThemeArg(themeArg: string): AdventureTheme | 'all' | null {
  if (themeArg === 'all') return 'all';
  
  const validThemes: AdventureTheme[] = ['space', 'mythical', 'ancient', 'developer'];
  
  if (validThemes.includes(themeArg as AdventureTheme)) {
    return themeArg as AdventureTheme;
  }
  
  const themeNum = parseInt(themeArg);
  if (themeNum >= 1 && themeNum <= 4) {
    return validThemes[themeNum - 1];
  }
  
  return null;
}
  • This function demonstrates robust argument parsing, supporting both names and numeric IDs.
  • The use of a predefined array (validThemes) ensures only supported themes are accepted.
  • The fallback to null handles invalid inputs gracefully, maintaining system stability.
  • This approach highlights the importance of clear and flexible user input handling.
generateThemeCSS(theme: AdventureTheme, outputDir: string): void {
  const cssFiles = [
    path.join(this.themesDir, 'base.css'),
    path.join(this.themesDir, 'homepage.css'), 
    path.join(this.themesDir, 'animations.css'),
    path.join(this.themesDir, `${theme}.css`)
  ];
  
  let combinedCSS = '';
  
  cssFiles.forEach(cssFile => {
    if (fs.existsSync(cssFile)) {
      combinedCSS += fs.readFileSync(cssFile, 'utf8') + '\\n\\n';
    }
  });
  
  const cssPath = path.join(outputDir, 'assets', 'theme.css');
  fs.writeFileSync(cssPath, combinedCSS);
}
  • This method dynamically combines multiple CSS files into a single stylesheet.
  • The use of fs.existsSync ensures non-existent files are skipped, preventing errors.
  • Combining CSS files enhances performance by reducing the number of HTTP requests.
  • The modular structure supports easy addition or modification of theme-specific styles.

export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];

  const unique = new Set<string>();
  const stack: any[] = [parsed];

  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;

    if (typeof node.path === 'string' && node.path.trim()) {
      const rel = node.path.trim();
      if (fs.existsSync(path.resolve(projectPath, rel))) {
        unique.add(rel);
      }
    }

    const children = Array.isArray(node) ? node : Object.values(node);
    children.forEach(child => {
      if (child && typeof child === 'object') stack.push(child);
    });
  }

  return Array.from(unique);
}
  • This function demonstrates recursive traversal for extracting file paths.
  • The use of a Set ensures all paths are unique, avoiding duplicates.
  • Validating paths with fs.existsSync prevents errors from invalid references.
  • This approach showcases robust data extraction techniques for dynamic systems.
