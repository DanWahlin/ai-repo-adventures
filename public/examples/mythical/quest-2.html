private setupHandlers() {
  this.server.setRequestHandler(ListToolsRequestSchema, async () => {
    const toolList = Object.entries(tools).map(([name, tool]) => ({
      name,
      description: tool.description,
      inputSchema: zodToJsonSchema(tool.schema, { 
        target: 'jsonSchema7',
        $refStrategy: 'none'
      })
    }));

    return { tools: toolList };
  });

  this.server.setRequestHandler(CallToolRequestSchema, async (request) => {
    try {
      const { name, arguments: args } = request.params;

      if (!(name in tools)) {
        throw new McpError(ErrorCode.MethodNotFound, `Unknown tool: ${name}`);
      }

      const tool = tools[name as keyof typeof tools];
      
      const validationResult = tool.schema.safeParse(args);
      if (!validationResult.success) {
        const errorMessages = validationResult.error.issues.map((err) => 
          `${err.path.join('.')}: ${err.message}`
        ).join(', ');
        throw new McpError(ErrorCode.InvalidParams, `Invalid parameters: ${errorMessages}`);
      }

      return await tool.handler(validationResult.data as any);

    } catch (error) {
      if (error instanceof McpError) {
        throw error;
      }
      throw new McpError(
        ErrorCode.InternalError,
        `Tool execution failed: ${error instanceof Error ? error.message : String(error)}`
      );
    }
  });
}
  • This code dynamically registers tools and validates input parameters using Zod schemas.
  • The ListToolsRequestSchema handler provides a manifest of tools, ensuring transparency for clients.
  • Validation prevents invalid or malicious tool executions, enhancing system security.
  • Error handling ensures graceful recovery from tool execution failures.
  • The decoupled architecture allows easy addition of new tools without modifying core logic.
async run() {
  const transport = new StdioServerTransport();
  await this.server.connect(transport);
  console.error('Repo Adventure MCP server running on stdio');
  
  const projectPath = process.cwd();
  console.error(`Pre-generating repomix content for project at ${projectPath}...`);
  repoAnalyzer.preGenerate(projectPath);
}
  • The run method connects the transport layer for communication and initializes the server.
  • Pre-generating content warms up caches, reducing latency during user interactions.
  • The use of process.cwd() ensures the server operates within the current project directory.
  • Background processing allows the server to remain responsive while pre-generating content.
  • This approach demonstrates proactive optimization techniques for interactive systems.
async main() {
  try {
    const server = new RepoAdventureServer();
    
    ['SIGINT', 'SIGTERM'].forEach(sig => 
      process.on(sig as NodeJS.Signals, gracefulShutdown)
    );
    
    process.on('unhandledRejection', (reason) => {
      console.error('Unhandled promise rejection:', reason);
      console.error('MCP server continuing to run. Please report this error.');
    });
    
    await server.run();
  } catch (error) {
    console.error('Fatal error starting MCP server:', error);
    process.exit(1);
  }
}
  • The main function orchestrates the server startup and handles shutdown signals gracefully.
  • Signal handling ensures the server cleans up resources during termination.
  • Logging unhandled promise rejections provides diagnostic information without crashing the server.
  • The separation of concerns between initialization, signal handling, and error management enhances maintainability.
  • This design ensures robust operation even under unexpected conditions.
