async generateRepomixContext(projectPath: string, options: RepomixOptions = {}): Promise<string> {
  this.validateProjectPath(projectPath);
  
  const cacheKey = `${path.resolve(projectPath)}:${JSON.stringify(options)}`;
  const cached = this.cache.get(cacheKey);
  if (cached && (Date.now() - cached.timestamp) < REPOMIX_CACHE_TTL) {
    return cached.content;
  }
  
  const cliOptions: CliOptions = {
    style: options.style || 'markdown',
    compress: options.compress !== false,
    ignore: ['node_modules', 'dist', 'build', '.git'],
    removeComments: true,
    removeEmptyLines: true,
    noDirectoryStructure: true
  };

  const context = await this.captureRepomixStdout(['.'], projectPath, cliOptions);
  this.cache.set(cacheKey, { content: context, timestamp: Date.now() });
  return context;
}
  • This method analyzes the codebase, using caching to avoid redundant computations.
  • It validates project paths to prevent security vulnerabilities.
  • CLI options ensure efficient content extraction, removing unnecessary lines and comments.
  • Subprocess invocation captures repomix output securely, with timeout protection.
private async captureRepomixStdout(directories: string[], cwd: string, options: CliOptions): Promise<string> {
  return new Promise((resolve, reject) => {
    const args = [...directories, '--stdout', '--style', options.style || 'markdown'];
    if (options.compress) args.push('--compress');

    const repomix = spawn('npx', ['repomix', ...args], { cwd, stdio: ['pipe', 'pipe', 'pipe'] });
    let stdout = '';
    let stderr = '';
    let stdoutSize = 0;

    const timeout = setTimeout(() => {
      repomix.kill('SIGTERM');
      setTimeout(() => repomix.kill('SIGKILL'), REPOMIX_GRACEFUL_TIMEOUT);
      reject(new Error(`Repomix subprocess timed out after ${REPOMIX_SUBPROCESS_TIMEOUT}ms`));
    }, REPOMIX_SUBPROCESS_TIMEOUT);

    repomix.stdout.on('data', (data) => {
      const chunk = data.toString();
      stdoutSize += chunk.length;
      if (stdoutSize > REPOMIX_MAX_BUFFER_SIZE) {
        repomix.kill('SIGKILL');
        reject(new Error(`Repomix output too large (${stdoutSize} bytes)`));
      }
      stdout += chunk;
    });

    repomix.on('close', (code) => {
      clearTimeout(timeout);
      if (code === 0) resolve(stdout);
      else reject(new Error(`Repomix failed with exit code ${code}: ${stderr}`));
    });
  });
}
  • This code securely invokes the repomix subprocess, handling stdout and stderr streams.
  • Timeout protection prevents hanging processes, ensuring reliability.
  • Memory constraints protect against excessive output size, avoiding resource exhaustion.
  • Graceful and forceful termination ensures subprocess cleanup.

async generateResponse(prompt: string, options?: LLMRequestOptions): Promise<LLMResponse> {
  await this.applyThrottling();

  try {
    const requestParams = this.buildRequestParams(prompt, options);
    const completion = await this.executeRequest(requestParams);
    const content = this.validateResponse(completion);

    this.logTokenUsage(completion, options?.context);
    this.onSuccessfulRequest();

    return { content };
  } catch (error) {
    const rateLimitInfo = this.detectRateLimitType(error);
    if (rateLimitInfo.type !== RateLimitType.NONE) {
      this.activateThrottling(error);
      throw new RateLimitError(rateLimitInfo.type, rateLimitInfo.waitSeconds, rateLimitInfo.message, error);
    }
    throw new Error(`LLM request failed: ${error.message}`);
  }
}
  • This method constructs and executes LLM requests, validating responses for correctness.
  • Adaptive throttling ensures stability under rate-limit constraints.
  • Detailed error handling provides actionable feedback for debugging failed requests.
  • Token usage logging helps monitor resource consumption.
private activateThrottling(error: any): void {
  this.isThrottling = true;
  const rateLimitInfo = this.detectRateLimitType(error);

  const suggestedDelay = rateLimitInfo.waitSeconds * 1000;
  this.throttleDelay = Math.min(suggestedDelay, this.MAX_THROTTLE_DELAY);

  console.error(`Throttling activated: ${(this.throttleDelay / 1000).toFixed(1)}s delay`);
}
  • This code dynamically adjusts throttling delays based on detected rate limits.
  • It ensures compliance with provider constraints, avoiding request failures.
  • Gradual delay reduction promotes recovery from rate-limited states.
