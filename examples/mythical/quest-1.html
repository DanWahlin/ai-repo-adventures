private loadTemplate(templateName: string): string {
  if (this.templateCache.has(templateName)) {
    return this.templateCache.get(templateName)!;
  }

  const templatePath = path.join(this.templatesDir, templateName);
  
  try {
    const content = fs.readFileSync(templatePath, 'utf-8');
    this.templateCache.set(templateName, content);
    return content;
  } catch (error) {
    throw new Error(`Failed to load template ${templateName}: ${error}`);
  }
}
  • This code caches templates to reduce redundant file reads, improving performance.
  • The templateCache map ensures that frequently used templates are stored in memory.
  • Error handling prevents runtime failures when templates are missing or inaccessible.
  • The use of fs.readFileSync ensures synchronous loading, critical for predictable rendering.
private replacePlaceholders(template: string, variables: TemplateVariables): string {
  let result = template;
  
  for (const [key, value] of Object.entries(variables)) {
    const placeholder = `{{${key}}}`;
    result = result.replace(new RegExp(placeholder, 'g'), value || '');
  }
  
  return result;
}
  • This function dynamically replaces placeholders (e.g., {{CONTENT}}) with actual values.
  • The use of RegExp ensures that all occurrences of a placeholder are replaced.
  • This approach supports modular template design, enabling flexible content injection.
  • It demonstrates the importance of variable substitution in generating personalized HTML.
renderPage(contentTemplate: string, variables: TemplateVariables): string {
  const baseTemplate = this.loadTemplate('base-template.html');
  const contentHtml = this.loadTemplate(contentTemplate);
  
  const renderedContent = this.replacePlaceholders(contentHtml, variables);
  
  const pageVariables = {
    ...variables,
    CONTENT: renderedContent
  };
  
  return this.replacePlaceholders(baseTemplate, pageVariables);
}
  • Combines base and content templates to generate complete HTML pages.
  • Injects rendered content into the base template using the CONTENT variable.
  • Demonstrates the layering of templates for hierarchical content rendering.
  • This method highlights the importance of modular design in dynamic HTML generation.

copyImages(outputDir: string, isMultiTheme: boolean = false): void {
  const sourceImagesDir = path.join(this.sourceDir, 'assets', 'images');
  const targetImagesDir = path.join(outputDir, 'assets', 'images');

  try {
    if (fs.existsSync(sourceImagesDir)) {
      fs.mkdirSync(targetImagesDir, { recursive: true });
      const imageFiles = fs.readdirSync(sourceImagesDir);
      imageFiles.forEach(file => {
        const sourcePath = path.join(sourceImagesDir, file);
        const targetPath = path.join(targetImagesDir, file);
        fs.copyFileSync(sourcePath, targetPath);
      });
    }
  } catch (error) {
    console.log(chalk.yellow('‚ö†Ô∏è Warning: Could not copy images from source directory'));
  }
}
  • Copies images from source directories to theme-specific output directories.
  • Ensures directories are created recursively using fs.mkdirSync.
  • Demonstrates defensive programming with error handling for missing source directories.
  • Supports multi-theme generation by toggling the isMultiTheme flag.
copyQuestNavigator(outputDir: string): void {
  const templatesDir = path.join(this.sourceDir, 'templates');
  const cssSource = path.join(templatesDir, 'quest-navigator.css');
  const jsSource = path.join(templatesDir, 'quest-navigator.js');

  const assetsDir = path.join(outputDir, 'assets');
  fs.mkdirSync(assetsDir, { recursive: true });

  if (fs.existsSync(cssSource)) {
    fs.copyFileSync(cssSource, path.join(assetsDir, 'quest-navigator.css'));
  }
  if (fs.existsSync(jsSource)) {
    fs.copyFileSync(jsSource, path.join(assetsDir, 'quest-navigator.js'));
  }
}
  • Transfers navigation-related assets (CSS and JS) to theme-specific directories.
  • Ensures compatibility with single-theme adventures by consolidating assets.
  • Demonstrates modular asset management for consistent navigation styling.
copyGlobalAssets(outputDir: string): void {
  const globalAssetsDir = path.join(outputDir, 'assets', 'shared');
  fs.mkdirSync(globalAssetsDir, { recursive: true });

  const globalImages = ['github-mark.svg', 'github-mark-white.svg'];
  globalImages.forEach(file => {
    const sourcePath = path.join(this.sourceDir, 'assets', 'shared', file);
    if (fs.existsSync(sourcePath)) {
      const targetPath = path.join(globalAssetsDir, file);
      fs.copyFileSync(sourcePath, targetPath);
    }
  });
}
  • Copies shared assets like logos and icons to a global directory for multi-theme adventures.
  • Demonstrates the importance of centralized asset management for cross-theme consistency.
  • Ensures shared assets are accessible from all generated themes.
