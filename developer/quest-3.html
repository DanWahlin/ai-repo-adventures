export function loadAdventureConfig(projectPath: string): string | null {
  const configPath = path.join(projectPath, 'adventure.config.json');
  return readFileIfExists(configPath);
}

export function parseAdventureConfig(projectPath: string): unknown | null {
  const raw = loadAdventureConfig(projectPath);
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch {
    return null;
  }
}

export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];

  const unique = new Set<string>();
  const stack: any[] = [parsed];

  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;

    if (typeof node.path === 'string' && fs.existsSync(path.resolve(projectPath, node.path.trim()))) {
      unique.add(node.path.trim());
    }

    (Array.isArray(node) ? node : Object.values(node)).forEach(child => {
      if (child && typeof child === 'object') stack.push(child);
    });
  }

  return Array.from(unique);
}
  • loadAdventureConfig: Loads the config as plain text; simplifies file path concatenation and reading logic.
  • parseAdventureConfig: Handles parsing errors gracefully, isolating downstream components from invalid configurations.
  • extractUniqueFilePaths: Ensures only valid real-world paths are included, creating a robust data pipeline.

packages/core/src/adventure/adventure-manager.ts

async initializeAdventure(projectInfo: ProjectInfo, theme: AdventureTheme, projectPath?: string): Promise<string> {
  this.state.reset();
  this.state.projectInfo = projectInfo;
  this.state.currentTheme = theme;
  this.state.projectPath = projectPath || process.cwd();

  if (theme === 'custom' && customThemeData) {
    this.storyGenerator.setCustomTheme(customThemeData);
  }

  const storyResponse = await this.storyGenerator.generateStoryAndQuests(projectInfo, theme, this.state.projectPath);

  this.state.title = storyResponse.title;
  this.state.story = storyResponse.story;

  this.state.quests = this.mergeQuestFilesFromConfig(storyResponse.quests, this.state.projectPath);
  return this.formatStoryWithQuests({ ...storyResponse, quests: this.state.quests });
}

exploreQuest(choice: string): Promise<AdventureResult> {
  const quest = this.findQuest(this.validateAndSanitizeChoice(choice));
  if (!quest) return this.createNotFoundResult();

  return this.executeQuest(quest);
}

private mergeQuestFilesFromConfig(quests: Quest[], projectPath: string | undefined): Quest[] {
  const config = parseAdventureConfig(projectPath || '');
  const adventureQuests = config?.adventure?.quests ?? [];
  
  return quests.map((q, i) => ({
    ...q,
    codeFiles: adventureQuests[i]?.files?.map((f: any) => f.path) ?? []
  }));
}
  • initializeAdventure: Integrates StoryGenerator and config files dynamically to prepare quests and associated context.
  • exploreQuest: Dynamically invokes a quest handler or retrieves cached results.
  • mergeQuestFilesFromConfig: Aligns generated quest structure with user-defined file references.

async generateStoryAndQuests(projectInfo: ProjectInfo, theme: AdventureTheme, projectPath?: string): Promise<StoryResponse> {
  const adventureGuidance = formatAdventureConfigForPrompt(projectPath || '');
  const prompt = loadStoryGenerationPrompt({
    theme,
    repomixContent: projectInfo.repomixContent || '',
    adventureGuidance,
    customThemeData: this.customThemeData
  });
  const response = await this.llmClient.generateResponse(prompt);
  return parseMarkdownToStoryResponse(response.content.trim());
}

async generateQuestContent(config: QuestGenerationConfig): Promise<QuestContent> {
  const questContext = formatAdventureConfigForPrompt(this.projectPath || '');
  const prompt = loadQuestContentPrompt({
    theme: config.theme,
    codeContent: config.codeContent,
    questPosition: config.questPosition,
    totalQuests: config.totalQuests,
    adventureGuidance: questContext,
  });
  const response = await this.llmClient.generateResponse(prompt);
  return {
    adventure: response.content.trim(),
    codeSnippets: [],
    hints: []
  };
}

function parseMarkdownToStoryResponse(markdownContent: string): StoryResponse {
  const tokens = marked.lexer(markdownContent);
  const title = tokens.find(t => t.type === 'heading' && t.depth === 1)?.text || 'Adventure';
  const story = tokens.filter(t => t.type === 'paragraph').map(t => t.text).join('\n\n');
  return { title, story, quests: [] };
}
  • generateStoryAndQuests: Synthesizes the storytelling setup by combining project data, themes, and language model processing.
  • generateQuestContent: Enriches quests with refined markdown content and contextualized details.
  • parseMarkdownToStoryResponse: Transforms markdown into structured objects; the key to extracting meaningful quest data.
