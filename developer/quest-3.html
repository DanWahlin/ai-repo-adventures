async initializeAdventure(
  projectInfo: ProjectInfo, 
  theme: AdventureTheme, 
  projectPath?: string,
  customThemeData?: CustomThemeData
): Promise<string> {
  this.state.reset();
  this.state.projectInfo = projectInfo;
  this.state.currentTheme = theme;
  this.state.projectPath = projectPath || process.cwd();
  
  if (theme === 'custom' && customThemeData) {
    this.storyGenerator.setCustomTheme(customThemeData);
  }

  const storyResponse = await this.storyGenerator.generateStoryAndQuests(projectInfo, theme, this.state.projectPath);
  
  this.state.title = storyResponse.title;
  this.state.story = storyResponse.story;

  this.state.quests = this.mergeQuestFilesFromConfig(storyResponse.quests, this.state.projectPath);
  this.state.quests = this.enforceConfigQuestCount(this.state.quests, this.state.projectPath);

  return this.formatStoryWithQuests({ ...storyResponse, quests: this.state.quests });
}
  • Initializes state for a new adventure, ensuring no residual settings from prior explorations.
  • Incorporates custom theme data for flexibility.
  • Responsible for merging external adventure.config.json settings into generated quests for config-guided exploration.
  • Finalizes the story and quest structure by enforcing quest count limits.
async exploreQuest(choice: string): Promise<AdventureResult> {
  const sanitizedChoice = this.validateAndSanitizeChoice(choice);

  if (this.isProgressRequest(sanitizedChoice)) {
    return this.getProgress();
  }

  const quest = this.findQuest(sanitizedChoice);
  if (!quest) {
    return this.createNotFoundResult();
  }

  return await this.executeQuest(quest);
}
  • Implements user input validation and sanitization to prevent invalid quest selection.
  • Differentiates between quest exploration and progress queries based on user intent.
  • Invokes quest execution methods, ensuring cached data reuse for already-completed quests.
private mergeQuestFilesFromConfig(quests: Quest[], projectPath: string | undefined): Quest[] {
  if (!projectPath) return quests;
  
  const config = parseAdventureConfig(projectPath);
  if (!config || typeof config !== 'object') return quests;

  const adventure = (config as any).adventure;
  if (!adventure || !Array.isArray(adventure.quests)) return quests;

  const configQuestFiles = new Map<string, string[]>();
  adventure.quests.forEach(configQuest => {
    if (configQuest.title && Array.isArray(configQuest.files)) {
      const filePaths = configQuest.files.filter((f: any) => f.path).map((f: any) => f.path);
      configQuestFiles.set(configQuest.title.toLowerCase(), filePaths);
    }
  });

  return quests.map((quest, index) => {
    if (index < adventure.quests.length && adventure.quests[index].files) {
      quest.codeFiles = adventure.quests[index].files.map((f: any) => f.path).filter(Boolean);
      return quest;
    }

    const questTitleLower = quest.title.toLowerCase();
    for (const [configTitle, files] of configQuestFiles.entries()) {
      if (questTitleLower.includes(configTitle) || configTitle.includes(questTitleLower)) {
        quest.codeFiles = files;
        return quest;
      }
    }

    return quest;
  });
}
  • Iterates over adventure.config to build a mapping between quest titles and file paths.
  • Relates quests to corresponding files for context-rich exploration targets.
  • Prioritizes strict order-based matching, followed by fallbacks using case-insensitive comparisons.

export function parseAdventureConfig(projectPath: string): unknown | null {
  const raw = loadAdventureConfig(projectPath);
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch {
    return null;
  }
}
  • Ensures robust handling of missing/invalid adventure.config.json files.
  • Features precise JSON parsing and error catching to avoid runtime interruptions.
  • Returns a clean structure only if the file is successfully read and valid JSON exists.
export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];

  const unique = new Set<string>();
  const stack: any[] = [parsed];

  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;

    if (typeof node.path === 'string' && fs.existsSync(path.resolve(projectPath, node.path.trim()))) {
      unique.add(node.path.trim());
    }

    (Array.isArray(node) ? node : Object.values(node)).forEach(child => {
      if (child && typeof child === 'object') stack.push(child);
    });
  }

  return Array.from(unique);
}
  • Combines breadth-first parsing of config files with filesystem validation to extract valid paths only.
  • Handles nested structures robustly without crashing on invalid intermediary nodes.
  • Outputs a clean list of file paths, ready for inclusion in quest generation.
export function formatAdventureConfigForPrompt(projectPath: string): string {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') {
    return '';
  }

  const adventure = (parsed as any).adventure;
  if (!adventure || !Array.isArray(adventure.quests)) {
    return '';
  }

  let formatted = `## Quest Structure\n`;
  adventure.quests.forEach(quest => {
    if (quest.title && Array.isArray(quest.files)) {
      formatted += `### ${quest.title}\nFiles: ${quest.files.map((f: any) => f.path).filter(Boolean).join(', ')}\n`;
    }
  });

  return formatted;
}
  • Reduces verbose config descriptions to optimized summaries, aiding LLM prompt efficiency.
  • Focuses on quest titles, file paths, and function highlights for succinct yet informative formatting.
  • Ensures readability and avoids overflow issues when generating LLM-based guidance.

async generateStoryAndQuests(projectInfo: ProjectInfo, theme: AdventureTheme, projectPath?: string): Promise<StoryResponse> {
  const prompt = loadStoryGenerationPrompt({
    theme,
    repomixContent: projectInfo.repomixContent || 'No repomix content available',
    adventureGuidance: formatAdventureConfigForPrompt(projectPath || ''),
    ...(extractCustomInstructions(projectPath || '') ? { customInstructions: extractCustomInstructions(projectPath || '') } : {})
  });

  const response = await this.llmClient.generateResponse(prompt, { maxTokens: LLM_MAX_TOKENS_STORY });

  const cleanContent = response.content.startsWith('```markdown') 
    ? response.content.replace(/^```markdown\s*/, '').replace(/\s*```$/, '') 
    : response.content.trim();

  return parseMarkdownToStoryResponse(cleanContent);
}
  • Synthesizes prompts by combining project data, custom instructions, and optimized templates.
  • Validates large LLM responses, ensuring consistent markdown-based extraction and parsing.
  • Converts raw LLM-generated data into structured objects for downstream processing.
async generateQuestContent(config: QuestGenerationConfig): Promise<QuestContent> {
  const { quest, theme, codeContent } = config;

  const prompt = loadQuestContentPrompt({
    theme,
    adventureTitle: quest.title,
    codeContent,
    storyContent: this.currentStoryContent || '',
    adventureGuidance: formatAdventureConfigForPrompt(this.projectPath || ''),
    ...(extractCustomInstructions(this.projectPath || '') && { customInstructions: extractCustomInstructions(this.projectPath || '') })
  });

  const response = await this.llmClient.generateResponse(prompt, { maxTokens: LLM_MAX_TOKENS_QUEST });

  const cleanContent = response.content.replace(/^```markdown\s*/, '').replace(/\s*```$/, '').trim();

  return {
    adventure: cleanContent,
    fileExploration: '',
    codeSnippets: [],
    hints: []
  };
}
  • Encapsulates quest-specific code files and generates theme-compliant content via LLM prompts.
  • Handles complex contextual inputs such as story backgrounds, code highlights, and config-derived data.
  • Ensures clean markdown-based outputs with no extraneous meta-commentary.
async generateCompletionSummary(quest: Quest, theme: AdventureTheme, progress: number, total: number): Promise<string> {
  const prompt = loadCompletionPrompt({
    themeDescription: theme,
    adventureTitle: quest.title,
    progress,
    total,
    percentComplete: Math.round((progress / total) * 100),
    vocabularyHint: theme === 'developer' ? 'Use professional technical documentation language' : ''
  });

  const response = await this.llmClient.generateResponse(prompt);
  return response.content.trim().replace(/^["']|["']$/g, '');
}
  • Builds detailed prompts for summarizing achievement in a theme-specific style.
  • Ensures calculated progress percentages are included for motivational feedback.
  • Strips unnecessary formatting elements from LLM responses.
