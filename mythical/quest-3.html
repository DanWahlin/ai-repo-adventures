/**
 * Loads the raw adventure config text if present.
 * Pure file read - no parsing here.
 */
export function loadAdventureConfig(projectPath: string): string | null {
  const configPath = path.join(projectPath, ADVENTURE_CONFIG_FILE);
  return readFileIfExists(configPath);
}
  • This function attempts to load the adventure.config.json file without parsing its contents.
  • Returns raw JSON text or null if the file is missing or inaccessible, ensuring graceful error handling.
  • Its lightweight design makes it a dependable entry point for configuration tasks.
/**
 * Parses the adventure config into an object (or null on error/missing).
 * Single point of JSON parsing and validation.
 */
export function parseAdventureConfig(projectPath: string): unknown | null {
  const raw = loadAdventureConfig(projectPath);
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch {
    return null;
  }
}
  • This code parses the configuration using JSON.parse, catching syntax errors that would otherwise disrupt execution.
  • Ensures that corrupted or non-existent configurations return safely as null.
  • Forms the base for higher-level logic requiring structured config data.
/**
 * Extracts all unique, existing file paths referenced by "path" fields anywhere in the config.
 */
export function extractUniqueFilePaths(projectPath: string): string[] {
  const parsed = parseAdventureConfig(projectPath);
  if (!parsed || typeof parsed !== 'object') return [];
  const unique = new Set<string>();
  const stack: any[] = [parsed];
  while (stack.length) {
    const node = stack.pop();
    if (!node || typeof node !== 'object') continue;
    if (typeof node.path === 'string' && node.path.trim()) {
      const rel = node.path.trim();
      if (fs.existsSync(path.resolve(projectPath, rel))) {
        unique.add(rel);
      }
    }
    const children = Array.isArray(node) ? node : Object.values(node);
    children.forEach(child => child && typeof child === 'object' && stack.push(child));
  }
  return Array.from(unique);
}
  • This function traverses nested objects to locate and validate file paths from the configuration.
  • Eliminates duplicates using a Set and verifies path existence with fs.existsSync.
  • Its recursive structure is robust for deeply nested configurations.

/**
 * Initialize the adventure with project context and generate story + quests
 */
async initializeAdventure(
  projectInfo: ProjectInfo, 
  theme: AdventureTheme, 
  projectPath?: string,
  customThemeData?: CustomThemeData
): Promise<string> {
  // Reset state for new adventure
  this.state.reset();
  this.state.projectInfo = projectInfo;
  this.state.currentTheme = theme;
  this.state.projectPath = projectPath || process.cwd();
  if (theme === 'custom' && customThemeData) {
    this.storyGenerator.setCustomTheme(customThemeData);
  }
  const storyResponse = await this.storyGenerator.generateStoryAndQuests(projectInfo, theme, this.state.projectPath);
  this.state.title = storyResponse.title;
  this.state.story = storyResponse.story;
  this.state.quests = this.mergeQuestFilesFromConfig(storyResponse.quests, this.state.projectPath);
  this.state.quests = this.enforceConfigQuestCount(this.state.quests, this.state.projectPath);
  return this.formatStoryWithQuests({ ...storyResponse, quests: this.state.quests });
}
  • This function establishes a new adventure by clearing state, setting context, and invoking quest generation.
  • Custom themes and configurations are integrated seamlessly using optional arguments.
  • Enforces constraints and combines data to crate a harmonious narrative framework.
